country_event = {
	id = RED_event.1
	title = RED_event.1.t
	desc = RED_event.1.d
	picture = GFX_report_event_german_troops

	is_triggered_only = yes

	trigger = {
		NOT = { has_war_with = ROOT }
	}

	option = {
		name = RED_event.1.a
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		ROOT = { country_event = { id = RED_event.2 days = 1 } }
		effect_tooltip = {
			ROOT = {
				transfer_state = 281
				add_state_core = 281
			}
			FROM = {
				remove_state_claim = 281
			}
		}
	}

	option = {
		name = RED_event.1.b
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				strength_ratio = {
					tag = FROM
					ratio > 0.7
				}
			}
			modifier = {
				factor = 10
				any_other_country = {
					is_major = yes
					is_in_faction_with = ROOT
				}
			}
		}
		FROM = {
			country_event = { id = RED_event.3 days = 1 }
			if = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = yes
				}
				remove_from_faction = ROOT
			}
			if = {
				limit = {
					is_in_faction_with = ROOT
					is_faction_leader = no
				}
				random_other_country = {
					limit = {
						is_in_faction_with = FROM
						is_faction_leader = yes
					}
					remove_from_faction = FROM
				}
			}
		}
		effect_tooltip = {
			FROM = {
				declare_war_on = {
					target = ROOT
					type = take_state
					generator = { 281 }
				}
			}
		}
	}
}

country_event = {
	id = RED_event.3
	title = RED_event.3.t
	desc = RED_event.3.d
	picture = GFX_report_event_polish_tanks_02

	is_triggered_only = yes

	option = {
		name = RED_event.3.a
		if = {
			limit = {
				FROM = {
					is_subject = yes
					OVERLORD = {
						has_non_aggression_pact_with = ROOT
					}
				}
			}
			FROM = {
				OVERLORD = {
					diplomatic_relation = {
						country = ROOT
						relation = non_aggression_pact
						active = no
					}
				}
			}
		}
		if = {
			limit = {
				FROM = {
					has_non_aggression_pact_with = ROOT
				}
			}
			FROM = {
				diplomatic_relation = {
					country = ROOT
					relation = non_aggression_pact
					active = no
				}
			}
			every_other_country = {
				limit = {
					has_non_aggression_pact_with = ROOT
				}
				add_opinion_modifier = {
					target = ROOT
   					modifier = protest_action_strong
				}
			}
		}
		add_state_claim = 281
	}
}
