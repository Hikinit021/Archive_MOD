country_event = {
	id = ABY_event.19
	title = ABY_event.19.t
	desc = ABY_event.19.desc
	
	is_triggered_only = yes

	option = {
		name = ABY_event.19.a
		FROM = {
			every_unit_leader = {
				set_nationality = ABY
			}
		}
		annex_country = { target = FROM transfer_troops = yes }
		if = {
			limit = {
				FROM = { tag = HYA }
				HYA = {
					exists = no
				}
			}
			BAC = {
				every_unit_leader = {
					set_nationality = ABY
				}
			}
			annex_country = { target = BAC transfer_troops = yes }
			SUH = {
				every_unit_leader = {
					set_nationality = ABY
				}
			}
			annex_country = { target = SUH transfer_troops = yes }
			INK = {
				every_unit_leader = {
					set_nationality = ABY
				}
			}
			annex_country = { target = INK transfer_troops = yes }
		}
	}
}
